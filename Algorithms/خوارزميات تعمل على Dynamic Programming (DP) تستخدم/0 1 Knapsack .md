  int n = weights.length;
  List<List<int>> dp = List.generate(n + 1, (_) => List.filled(W + 1, 0));

  for (int i = 1; i <= n; i++) {
    for (int w = 0; w <= W; w++) {
      if (weights[i - 1] <= w) {
        dp[i][w] =
          dp[i - 1][w].clamp(0, double.infinity.toInt()).toInt().compareTo(
            values[i - 1] + dp[i - 1][w - weights[i - 1]]
          ) < 0
            ? values[i - 1] + dp[i - 1][w - weights[i - 1]]
            : dp[i - 1][w];

      }
    }
  }

  return dp[n][W];
}
